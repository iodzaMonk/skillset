export async function deleteObject(key: string) {
  if (!s3 || !BUCKET) {
    inMemoryBucket.delete(key);
    return;
  }

  const deleteCommand = new DeleteObjectCommand({
    Bucket: BUCKET,
    Key: key,
  });

  await s3.send(deleteCommand);
}
