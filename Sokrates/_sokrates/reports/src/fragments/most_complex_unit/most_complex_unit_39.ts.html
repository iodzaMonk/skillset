const removeCommentFromTree = (comments: Comment[], id: string): Comment[] =>
  comments
    .filter((comment) => comment.id !== id)
    .map((comment) =>
      comment.replies && comment.replies.length > 0
        ? {
            ...comment,
            replies: removeCommentFromTree(comment.replies, id),
          }
        : comment,
    );
