  async fetchBySlug(slug: string): Promise<ProductResponse> {
    if (!slug) {
      return {
        status: 400,
        body: { message: "Invalid product ID" },
      };
    }

    try {
      const product = await prisma.posts.findUnique({
        where: { id: slug },
        include: {
          users: {
            select: {
              id: true,
              name: true,
              country: true,
              email: true,
            },
          },
          reviews: {
            include: {
              users: {
                select: {
                  id: true,
                  name: true,
                  country: true,
                  email: true,
                },
              },
              replies: {
                select: {
                  id: true,
                  date: true,
                  text: true,
                },
              },
            },
            orderBy: { date: "desc" },
          },
        },
      });

      if (!product) {
        return {
          status: 404,
          body: { message: "Product not found" },
        };
      }

      return {
        status: 200,
        body: {
          ...product,
          image_url: "/no-image.svg",
        },
      };
    } catch (error) {
      if (error instanceof Prisma.PrismaClientKnownRequestError) {
        return {
          status: 400,
          body: { message: "Invalid product ID" },
        };
      }
      throw error;
    }
  }
